def map_event(e):
    if e in ["visualizacao_aula","visualizacao_prova","visualizacao_avaliacao",
             "visualizacao_metodologia_ativa","visualizacao_conteudo_ia","visualizacao_plano_aula"]:
        return 1
    elif e in ["download_aula","download_avaliacao","download_conteudo_ia","download_plano_aula"]:
        return 2
    elif e in ["compartilhamento_direto","botao_compartilhar_conquista_modal",
               "botao_compartilhar_conquista_completada"]:
        return 3
    elif e in ["criacao_plano_aula","botao_criar_plano_aula","rascunho_plano_aula",
               "prova_salva","criacao_anotacao_relatorio","criacao_turma_relatorio"]:
        return 4
    elif e in ["acesso_aba_conquistas","click_selo_conquista","click_link_missao_aulas",
               "click_link_missao_comunidade","click_link_missao_relatorio_turma",
               "click_link_missao_mari","click_link_missao_plano_de_aula","click_link_missao_blog"]:
        return 2
    else:
        return 1   # default

def leave_one_out_split(df, user_col="unique_id", item_col="id_aula"):
    # pega a ltima interao de cada user para TEST (pela ordem atual do DF)
    df = df.drop_duplicates([user_col, item_col])  # evita duplicatas exatas
    last_idx = df.groupby(user_col, sort=False).tail(1).index
    test_df  = df.loc[last_idx]
    train_df = df.drop(index=last_idx)
    # garante que user de test existe em train
    users_in_train = set(train_df[user_col].unique())
    test_df = test_df[test_df[user_col].isin(users_in_train)]
    return train_df, test_df

def build_reindexers(train_df, user_col="unique_id", item_col="id_aula"):
    users = train_df[user_col].unique()
    items = train_df[item_col].unique()
    user2idx = {u:i for i,u in enumerate(users)}
    item2idx = {it:j for j,it in enumerate(items)}
    return user2idx, item2idx

def map_index(df, user2idx, item2idx, user_col="unique_id", item_col="id_aula"):
    X = df.copy()
    X["user_idx"] = X[user_col].map(user2idx)
    X["item_idx"] = X[item_col].map(item2idx)
    # remove o que ficou fora do universo do treino
    X = X.dropna(subset=["user_idx","item_idx"])
    X["user_idx"] = X["user_idx"].astype(np.int64)
    X["item_idx"] = X["item_idx"].astype(np.int64)
    return X

def to_csr(df, n_users, n_items, rating_col="rating_log"):
    r = df[rating_col].astype(float).to_numpy()
    u = df["user_idx"].to_numpy()
    i = df["item_idx"].to_numpy()
    return coo_matrix((r, (u, i)), shape=(n_users, n_items)).tocsr()

def _coerce_users(users, n_users_model, train_csr, test_csr):
    if users is None:
        train_has = np.where(train_csr.getnnz(axis=1) > 0)[0]
        test_has  = np.where(test_csr.getnnz(axis=1)  > 0)[0]
        users = np.intersect1d(train_has, test_has)
    else:
        users = np.asarray(list(set(users)), dtype=np.int64).ravel()
        users = np.unique(users)
        train_has = np.where(train_csr.getnnz(axis=1) > 0)[0]
        test_has  = np.where(test_csr.getnnz(axis=1)  > 0)[0]
        users = np.intersect1d(users, np.intersect1d(train_has, test_has))
    return users[(users >= 0) & (users < n_users_model)]

def precision_recall_at_k(model, train_csr, test_csr, users=None, K=10):
    users = _coerce_users(users, model.user_factors.shape[0], train_csr, test_csr)
    precisions, recalls = [], []
    for u in users:
        test_items = test_csr[u].indices
        if len(test_items) == 0: 
            continue
        rec_ids, _ = model.recommend(u, train_csr[u], N=K,
                                     filter_already_liked_items=True,
                                     recalculate_user=False)
        hit = len(set(rec_ids) & set(test_items))
        precisions.append(hit / K)
        recalls.append(hit / len(test_items))
    return (float(np.mean(precisions)) if precisions else 0.0,
            float(np.mean(recalls)) if recalls else 0.0)

    # top-K itens mais populares (contagem de interaes no treino)
    item_pop = np.asarray(trainX.getnnz(axis=0)).ravel()
    topk = np.argsort(-item_pop)[:K]

    hits_total, rel_total = 0, 0
    for u in users:
        test_items = set(testX[u].indices)
        if not test_items:
            continue
        hits_total += len(test_items.intersection(topk))
        rel_total  += len(test_items)

    # Precision@K (aprox) e Recall@K do baseline
    P_pop = min(1.0, hits_total / (len(users) * K)) if len(users) else 0.0
    R_pop = (hits_total / rel_total) if rel_total else 0.0
    return P_pop, R_pop

def pr_at_k_model(model, train_weighted_ui, test_ui, users, K=10, filter_seen=True):
    precisions, recalls = [], []
    for u in users:
        ti = test_ui[u].indices
        if ti.size == 0: 
            continue
        seen = set(train_weighted_ui[u].indices) if filter_seen else set()
        recs, _ = model.recommend(
            userid=int(u),
            user_items=train_weighted_ui[u],   # usar BM25 useritem
            N=K + (len(seen) if filter_seen else 0),
            filter_items=list(seen) if filter_seen and seen else None,
            recalculate_user=False             # usa user_factors j treinados
        )
        recs = recs[:K]
        hit = len(set(recs) & set(ti))
        precisions.append(hit / K)
        recalls.append(hit / len(ti))
    return (float(np.mean(precisions)) if precisions else 0.0,
            float(np.mean(recalls))    if recalls    else 0.0)

def pr_at_k_pop(train_ui, test_ui, users, K=10):
    item_pop = train_ui.sum(axis=0).A1
    order = np.argsort(item_pop)[::-1]
    precisions, recalls = [], []
    for u in users:
        ti = test_ui[u].indices
        if ti.size == 0: 
            continue
        seen = set(train_ui[u].indices)
        recs = [int(it) for it in order if it not in seen][:K]
        hit = len(set(recs) & set(ti))
        precisions.append(hit / K); recalls.append(hit / len(ti))
    return (float(np.mean(precisions)) if precisions else 0.0,
            float(np.mean(recalls))    if recalls    else 0.0)

# ===== EASEr (Embarrassingly Shallow Autoencoder) =====
# Treina em X = train_w (BM25 useritem). Para >~10k itens, use cholesky/solve; aqui 2.7k  tranquilo.
import numpy as np

class EASEr:
    def __init__(self, l2=700.0):
        self.l2 = float(l2)
        self.B = None  # matriz itensitens

    def fit(self, X_ui):  # X_ui: CSR (usersitems), de preferncia ponderado (BM25/TFIDF)
        # Gram (itensitens)
        G = (X_ui.T @ X_ui).toarray().astype(np.float64)
        n_items = G.shape[0]
        G.flat[::n_items+1] += self.l2  # + I
        P = np.linalg.inv(G)            # (G + I)^{-1}
        # B = -P / diag(P); diag(B)=0
        diag = np.diag(P)
        B = -P / (diag[:, None] + 1e-12)
        np.fill_diagonal(B, 0.0)
        self.B = B.astype(np.float32)

    def recommend(self, userid, user_items, N=10, filter_items=None, recalculate_user=False, items=None):
        # scores = x_u @ B
        x = user_items.toarray().astype(np.float32)  # 1n_items
        scores = x @ self.B                          # 1n_items
        scores = scores.ravel()
        if filter_items is not None and len(filter_items) > 0:
            scores[np.array(filter_items, dtype=np.int64)] = -np.inf
        # top-N
        if N < len(scores):
            idx = np.argpartition(-scores, N)[:N]
            idx = idx[np.argsort(-scores[idx])]
        else:
            idx = np.argsort(-scores)
        return idx.astype(np.int32), scores[idx]

def build_item_cosine_dense(X_items_users, topk=None):
    """
    X: CSR (itemsusers). Retorna S densa (itemsitems) com cosseno.
    Se topk  fornecido, zera tudo exceto os topk mais similares por linha.
    """
    # normaliza cada item para norma L2=1 (linha)
    Xn = normalize(X_items_users, axis=1, norm='l2', copy=True)
    # produto (itemsusers)  (usersitems) = (itemsitems)
    S = (Xn @ Xn.T).toarray().astype("float32")
    # remove auto-similaridade
    np.fill_diagonal(S, 0.0)

    if topk is not None:
        k = int(min(topk, S.shape[1]-1))
        # mantm top-k por linha (in-place)
        for i in range(S.shape[0]):
            row = S[i]
            # ndices dos top-k (maiores)
            idx_top = np.argpartition(row, -k)[-k:]
            mask = np.ones_like(row, dtype=bool)
            mask[idx_top] = False
            row[mask] = 0.0
    return S

def fit_als_safe(UI_train, factors=128, reg=0.05, iters=20, use_cg=True, threads=0):
    """
    UI_train: CSR useritem
    Retorna (als, UI_w) j pronto p/ recommend.
    Garante: als.user_factors.shape[0] == n_users e als.item_factors.shape[0] == n_items
    """
    assert UI_train.shape[0] == train_csr.shape[0] and UI_train.shape[1] == train_csr.shape[1], \
        "UI_train deve casar com train_csr (useritem)."

    n_users, n_items = UI_train.shape

    # 1) prepara BM25 useritem e itemuser
    UI_w = bm25_weight(UI_train).astype("float32").tocsr()  # useritem
    IU_w = UI_w.T.tocsr()                                   # itemuser

    def _new():
        return AlternatingLeastSquares(
            factors=factors, regularization=reg, iterations=iters,
            use_cg=use_cg, num_threads=threads, calculate_training_loss=False
        )

    # 2) tenta com itemuser (o esperado pelo implicit)
    als = _new()

    # 3) checa; se no bater, tenta do outro lado
    if uf != n_users or vf != n_items:
        print("[WARN] Orientao no bateu. Tentando com UI (useritem)")

    cols = [c for c in cols if c in df.columns]
    df = df.loc[:, cols].copy()

    mask = df["userId"].notna() & df["itemId"].notna() \
           & (df["userId"].astype(str) != "") & (df["itemId"].astype(str) != "")
    df = df.loc[mask].copy()

    if has_ts:
        df = df.sort_values(["userId","data_inicio"])

# --- monta matriz de confiana: UI_conf = 1 +  * f(UI)
def build_conf(UI, mode='bin', alpha=40.0):
    if mode == 'bin':
        X = UI.copy().tocsr(); X.data[:] = 1.0
    elif mode == 'log':
        X = UI.copy().tocsr(); X.data = np.log1p(X.data).astype('float32')

import numpy as np
def ci_bootstrap(values, B=1000, alpha=0.05, seed=0):
    rng = np.random.default_rng(seed)
    boots = [np.mean(rng.choice(values, size=len(values), replace=True)) for _ in range(B)]
    lo, hi = np.quantile(boots, [alpha/2, 1-alpha/2])
    return float(np.mean(values)), float(lo), float(hi)
